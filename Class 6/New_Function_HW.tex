\section{New Function: plotbfactor}\label{new-function-plotbfactor}

Description:

The input to this funtion is the PDB ID of a protein of interest and
requires the installation of the bio3d package.

This function pulls up the PDB file of the protein, selects for the
atoms in chain A and with carbon alpha classification, and generates a
plot of the b factor data from these atoms.

The output of this function is a plot of the b factor data.

\begin{Shaded}
\begin{Highlighting}[]
\NormalTok{plotbfactor <-}\StringTok{ }\ControlFlowTok{function}\NormalTok{(x) \{}
  
  \CommentTok{#load the bio3d package}
  \KeywordTok{library}\NormalTok{(bio3d)}
  
  \CommentTok{#load the PDB coordinate file of the protein of interest}
\NormalTok{  s <-}\StringTok{ }\KeywordTok{read.pdb}\NormalTok{(x)}
  
  \CommentTok{#trim to chain A and Carbon alpha amino acid data}
\NormalTok{  s.chainA <-}\StringTok{ }\KeywordTok{trim.pdb}\NormalTok{(s, }\DataTypeTok{chain =} \StringTok{"A"}\NormalTok{, }\DataTypeTok{elety =} \StringTok{"CA"}\NormalTok{)}
  
  \CommentTok{#call the atom data then the b factor data from s1.chainA}
\NormalTok{  s.b <-}\StringTok{ }\NormalTok{s.chainA}\OperatorTok{$}\NormalTok{atom}\OperatorTok{$}\NormalTok{b}
  
  \CommentTok{#plot the b factor data}
\NormalTok{  bf_plot <-}\StringTok{ }\KeywordTok{plotb3}\NormalTok{(s.b, }\DataTypeTok{sse=}\NormalTok{s.chainA, }\DataTypeTok{typ=}\StringTok{"l"}\NormalTok{, }\DataTypeTok{ylab=}\StringTok{"Bfactor"}\NormalTok{)}
  
  \KeywordTok{return}\NormalTok{(bf_plot)}
\NormalTok{\}}
\end{Highlighting}
\end{Shaded}
